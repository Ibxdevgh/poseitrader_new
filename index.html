from poseitrader.trading import Strategy
from poseitrader.indicators import EMA, RSI
 
class MomentumStrategy(Strategy):
    # Triggered on every new bar
    def on_bar(self, bar):
        fast = EMA(bar.close, 12)
        slow = EMA(bar.close, 26)
        rsi = RSI(bar.close, 14)
 
        if fast > slow and rsi < 70:
            self.submit_order(
                MarketOrder(bar.instrument, 1.0)
            )
        elif fast < slow or rsi > 85:
            self.flatten(bar.instrument)
